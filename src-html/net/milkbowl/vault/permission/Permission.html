273    /**
274     * Add transient permission to a player.
275     * This operation adds a world-unspecific permission onto the player object in bukkit via Bukkit's permission interface.
276     * 
277     * @param player Player Object
278     * @param permission Permission node
279     * @return Success or Failure
280     */
281    public boolean playerAddTransient(Player player, String permission) {
282                for (PermissionAttachmentInfo paInfo : player.getEffectivePermissions()) {
283                        if (paInfo.getAttachment() != null && paInfo.getAttachment().getPlugin().equals(plugin)) {
284                                paInfo.getAttachment().setPermission(permission, true);
285                                return true;
286                        }
287                }
288
289                PermissionAttachment attach = player.addAttachment(plugin);
290                attach.setPermission(permission, true);
291
292                return true;
293    }
